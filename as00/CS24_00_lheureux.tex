\begin{enumerate}[leftmargin=*, label=\textit{\alph*}.]
    \item $ax + b$
    \item The average of the four numbers $w$, $x$, $y$, and $z$
    \item $a^2 + 2ab + b^2$ (Can you do it with one add box and one multiply box?)
\end{enumerate}
